<?php

namespace App\Services;

use Illuminate\Support\Facades\Log;
use Illuminate\Support\Facades\Http;

class SmsService
{
    public function send(string $phone, string $message): bool
    {
        // В режиме разработки — только логируем
        if (config('sms.debug')) {
            Log::info("SMS DEBUG → {$phone}: {$message}");
            return true;
        }

        return $this->sendViaSmsc($phone, $message);
    }

    private function sendViaSmsc(string $phone, string $message): bool
    {
        $response = Http::get('https://smsc.ru/sys/send.php', [
            'login'   => config('sms.smsc_login'),
            'psw'     => config('sms.smsc_password'),
            'phones'  => $phone,
            'mes'     => $message,
            'fmt'     => 3, // JSON ответ
            'charset' => 'utf-8',
        ]);

        if ($response->failed()) {
            Log::error('SMSC error: ' . $response->body());
            return false;
        }

        $data = $response->json();

        if (isset($data['error'])) {
            Log::error('SMSC error: ' . $data['error_code'] . ' — ' . $data['error']);
            return false;
        }

        return true;
    }
}

5. OtpController — обновить
php
<?php

namespace App\Http\Controllers\Api;

use App\Http\Controllers\Controller;
use App\Models\User;
use App\Services\SmsService;
use Illuminate\Http\Request;
use Illuminate\Support\Facades\Cache;

class OtpController extends Controller
{
    public function __construct(protected SmsService $sms) {}

    public function sendCode(Request $request)
    {
        $request->validate([
            'phone' => 'required|string|regex:/^\+?[0-9]{10,15}$/',
        ]);

        $phone = $request->phone;

        // Защита от спама — не чаще раза в минуту
        if (Cache::has("otp_lock:{$phone}")) {
            return response()->json([
                'message' => 'Подождите минуту перед повторной отправкой'
            ], 429);
        }

        $code = rand(1000, 9999);

        Cache::put("otp:{$phone}",      $code, now()->addMinutes(5));
        Cache::put("otp_lock:{$phone}", true,  now()->addMinute());

        $this->sms->send($phone, "Ваш код подтверждения: {$code}");

        // В debug режиме отдаём код в ответе
        $debug = config('sms.debug') ? ['debug_code' => $code] : [];

        return response()->json([
            'message' => 'Код отправлен',
            ...$debug,
        ]);
    }

    public function verifyCode(Request $request)
    {
        $request->validate([
            'phone' => 'required|string',
            'code'  => 'required|string|size:4',
        ]);

        $phone  = $request->phone;
        $cached = Cache::get("otp:{$phone}");

        if (!$cached || (string) $cached !== $request->code) {
            return response()->json(['message' => 'Неверный или истёкший код'], 422);
        }

        Cache::forget("otp:{$phone}");
        Cache::forget("otp_lock:{$phone}");

        $user = User::firstOrCreate(
            ['phone' => $phone],
            ['name'  => 'Пользователь', 'role' => 'user']
        );

        $token = $user->createToken('client-token')->plainTextToken;

        return response()->json(['user' => $user, 'token' => $token]);
    }
}

namespace App\Services;

use Illuminate\Support\Facades\Http;
use Illuminate\Support\Facades\Log;

class SmsService
{
    public function send(string $phone, string $message): bool
    {
        if (config('sms.debug')) {
            Log::info("SMS DEBUG → {$phone}: {$message}");
            return true;
        }

        $response = Http::get('https://smsc.ru/sys/send.php', [

            'phones'  => $phone,

        ]);

        if ($response->failed() || isset($response->json()['error'])) {
            Log::error('SMSC: ' . $response->body());
            return false;
        }

        return true;
    }
}

<?php
namespace App\Http\Controllers\Api\Admin;

use App\Http\Controllers\Controller;
use App\Models\User;
use Illuminate\Http\Request;
use Illuminate\Support\Facades\Hash;
use Illuminate\Validation\ValidationException;

class AdminAuthController extends Controller
{
    public function login(Request $request)
    {
        $request->validate([
            'email'    => 'required|email',
            'password' => 'required',
        ]);

        $user = User::where('email', $request->email)
                    ->where('role', 'admin')
                    ->first();

        if (!$user || !Hash::check($request->password, $user->password)) {
            throw ValidationException::withMessages([
                'email' => ['Неверные данные.'],
            ]);
        }

        $user->tokens()->delete();

        return response()->json([
            'user'  => $user,
            'token' => $user->createToken('admin-token')->plainTextToken,
        ]);
    }

    public function logout(Request $request)
    {
        $request->user()->currentAccessToken()->delete();
        return response()->json(['message' => 'Вышли успешно']);
    }
}
app/Http/Controllers/Api/ProductController.php:

<?php
namespace App\Http\Controllers\Api;

use App\Http\Controllers\Controller;
use App\Http\Resources\ProductResource;
use App\Models\Product;
use Illuminate\Http\Request;

class ProductController extends Controller
{
    public function index(Request $request)
    {
        $products = Product::with(['category', 'mainImage'])
            ->where('is_active', true)
            ->when($request->category, fn($q) =>
                $q->whereHas('category', fn($q) =>
                    $q->where('slug', $request->category)))
            ->when($request->search, fn($q) =>
                $q->where('name', 'like', "%{$request->search}%"))
            ->when($request->min_price, fn($q) =>
                $q->where('price', '>=', $request->min_price))
            ->when($request->max_price, fn($q) =>
                $q->where('price', '<=', $request->max_price))
            ->when($request->sort === 'price_asc',  fn($q) => $q->orderBy('price'))
            ->when($request->sort === 'price_desc', fn($q) => $q->orderByDesc('price'))
            ->when($request->sort === 'new',        fn($q) => $q->latest())
            ->paginate($request->per_page ?? 12);

        return ProductResource::collection($products);
    }

    public function show(string $slug)
    {
        $product = Product::with(['category', 'images'])
            ->where('slug', $slug)
            ->where('is_active', true)
            ->firstOrFail();

        return new ProductResource($product);
    }
}
app/Http/Controllers/Api/CartController.php:

php
<?php
namespace App\Http\Controllers\Api;

use App\Http\Controllers\Controller;
use App\Models\CartItem;
use App\Models\Product;
use Illuminate\Http\Request;

class CartController extends Controller
{
    private function cartResponse(Request $request)
    {
        $items = $request->user()
            ->cartItems()
            ->with('product.mainImage')
            ->get();

        return response()->json(['items' => $items]);
    }

    public function index(Request $request)
    {
        return $this->cartResponse($request);
    }

    public function store(Request $request)
    {
        $request->validate([
            'product_id' => 'required|exists:products,id',
            'quantity'   => 'integer|min:1',
        ]);

        $item = CartItem::firstOrNew([
            'user_id'    => $request->user()->id,
            'product_id' => $request->product_id,
        ]);

        $item->quantity = ($item->quantity ?? 0) + ($request->quantity ?? 1);
        $item->save();

        return $this->cartResponse($request);
    }

    public function update(Request $request, CartItem $cartItem)
    {
        $request->validate(['quantity' => 'required|integer|min:1']);
        $cartItem->update(['quantity' => $request->quantity]);
        return $this->cartResponse($request);
    }

<?php
namespace App\Http\Controllers\Api;

use App\Http\Controllers\Controller;
use App\Mail\OrderConfirmMail;
use App\Models\Order;
use Illuminate\Http\Request;
use Illuminate\Support\Facades\Mail;

class OrderController extends Controller
{
    public function index(Request $request)
    {
        $orders = $request->user()
            ->orders()
            ->with('items.product')
            ->latest()
            ->get();

        return response()->json($orders);
    }

    public function store(Request $request)
    {
        $request->validate([
            'phone'   => 'required|string',
            'address' => 'required|string',
            'comment' => 'nullable|string',
        ]);

        $items = $request->user()->cartItems()->with('product')->get();

        if ($items->isEmpty()) {
            return response()->json(['message' => 'Корзина пуста'], 422);
        }

        $total = $items->sum(fn($i) => $i->product->price * $i->quantity);

        $order = Order::create([
            'user_id' => $request->user()->id,
            'status'  => 'pending',
            'total'   => $total,
            'phone'   => $request->phone,
            'address' => $request->address,
            'comment' => $request->comment,
        ]);

        foreach ($items as $item) {
            $order->items()->create([
                'product_id' => $item->product_id,
                'quantity'   => $item->quantity,
                'price'      => $item->product->price,
            ]);
        }

        // Очищаем корзину
        $request->user()->cartItems()->delete();

        // Отправляем письмо если есть email
        if ($request->user()->email) {
            Mail::to($request->user()->email)
                ->send(new OrderConfirmMail($order->load('items.product')));
        }

        return response()->json($order->load('items.product'), 201);
    }

    public function show(Request $request, Order $order)
    {
        abort_unless($order->user_id === $request->user()->id, 403);
        return response()->json($order->load('items.product'));
    }

export const useApi = () => {
  const config = useRuntimeConfig()
  const token  = useCookie<string | null>('auth_token')

  const $api = $fetch.create({
    baseURL: config.public.apiBase as string,
    onRequest({ options }) {
      if (token.value) {
        options.headers = new Headers(options.headers as HeadersInit)
        options.headers.set('Authorization', `Bearer ${token.value}`)
        options.headers.set('Accept', 'application/json')
      }
    },
    async onResponseError({ response }) {
      if (response.status === 401) {
        token.value = null
        await navigateTo('/login')
      }
    },
  })

  return { $api }

export const useAuthStore = defineStore('auth', () => {
  const { $api } = useApi()

  const user  = ref<User | null>(null)
  const token = useCookie<string | null>('auth_token', {
    maxAge: 60 * 60 * 24 * 30,
    sameSite: 'lax',
  })

  const isLoggedIn = computed(() => !!token.value)
  const isAdmin    = computed(() => user.value?.role === 'admin')

  async function sendCode(phone: string) {
    return $api<{ message: string; debug_code?: number }>('/auth/send-code', {
      method: 'POST',
      body: { phone },
    })
  }

  async function verifyCode(phone: string, code: string) {
    const res = await $api<{ user: User; token: string }>('/auth/verify-code', {
      method: 'POST',
      body: { phone, code },
    })
    token.value = res.token
    user.value  = res.user
    await navigateTo('/')
  }

  async function adminLogin(email: string, password: string) {
    const res = await $api<{ user: User; token: string }>('/admin/auth/login', {
      method: 'POST',
      body: { email, password },
    })
    token.value = res.token
    user.value  = res.user
    await navigateTo('/admin')
  }

  async function logout() {
    const url = isAdmin.value ? '/admin/auth/logout' : '/auth/logout'
    try { await $api(url, { method: 'POST' }) } finally {
      token.value = null
      user.value  = null
      await navigateTo(isAdmin.value ? '/admin/login' : '/login')
    }
  }

  async function fetchMe() {
    if (!token.value) return
    try {
      user.value = await $api<User>('/me')
    } catch {
      token.value = null
    }
  }

  return { user, token, isLoggedIn, isAdmin, sendCode, verifyCode, adminLogin, logout, fetchMe }
})

import { defineStore } from 'pinia'

export const useCartStore = defineStore('cart', () => {
  const { $api } = useApi()

  const items   = ref<CartItem[]>([])
  const loading = ref(false)

  const totalCount = computed(() => items.value.reduce((s, i) => s + i.quantity, 0))
  const totalPrice = computed(() => items.value.reduce((s, i) => s + i.product.price * i.quantity, 0))
  const isEmpty    = computed(() => items.value.length === 0)

  const isInCart  = (id: number) => items.value.some(i => i.product.id === id)
  const getItem   = (id: number) => items.value.find(i => i.product.id === id)

  async function fetchCart() {
    loading.value = true
    try {
      const res = await $api<{ items: CartItem[] }>('/cart')
      items.value = res.items
    } finally {
      loading.value = false
    }
  }

  async function addItem(productId: number, quantity = 1) {
    const res = await $api<{ items: CartItem[] }>('/cart', {
      method: 'POST',
      body: { product_id: productId, quantity },
    })
    items.value = res.items
  }

  async function updateQuantity(itemId: number, quantity: number) {
    if (quantity <= 0) return removeItem(itemId)
    const res = await $api<{ items: CartItem[] }>(`/cart/${itemId}`, {
      method: 'PATCH',
      body: { quantity },
    })
    items.value = res.items
  }

  async function removeItem(itemId: number) {
    const res = await $api<{ items: CartItem[] }>(`/cart/${itemId}`, { method: 'DELETE' })
    items.value = res.items
  }

  async function clearCart() {
    await $api('/cart', { method: 'DELETE' })
    items.value = []
  }

  return { items, loading, totalCount, totalPrice, isEmpty, isInCart, getItem, fetchCart, addItem, updateQuantity, removeItem, clearCart }
})

<script setup lang="ts">
definePageMeta({ middleware: 'auth' })

const cart   = useCartStore()
const auth   = useAuthStore()
const { $api } = useApi()
const router = useRouter()
const error  = ref('')

const form = reactive({
  phone:   auth.user?.phone ?? '',
  address: '',
  comment: '',
})

async function handleOrder() {
  error.value = ''
  try {
    await $api('/orders', { method: 'POST', body: form })
    await cart.fetchCart()
    await router.push('/checkout/success')
  } catch (e: any) {
    error.value = e?.data?.message ?? 'Ошибка оформления'
  }
}
</script>

<template>
  <div class="checkout-page">
    <h1>Оформление заказа</h1>
    <p v-if="error" class="error">{{ error }}</p>

    <div class="order-form">
      <div class="field">
        <label>Телефон</label>
        <input v-model="form.phone" type="tel" />
      </div>
      <div class="field">

      </div>
